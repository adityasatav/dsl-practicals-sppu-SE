include <iostream>
#include <stack>
#include <string>
#define max 50
using namespace std;
char stk[max];
int top=-1;

class Expression
{
	public:
		bool Parenthesized(string exp) 
		{
     		for (char ch : exp) 
	 		{
      			 if (ch == '(' || ch == '[' || ch == '{') 
				{	
           				 // Push opening parentheses onto the stack
            			push(ch);
        		} 
				else if (ch == ')' || ch == ']' || ch == '}') 
				{
            		// Check if the stack is empty
            		if (top==-1) 
					{
              			  return false;
            		}

            		// Pop the top element from the stack
            		char topChar=pop();
            

            		// Check if the popped character matches the corresponding opening parenthesis
            		if ((ch == ')' && topChar != '(') ||
                		(ch == ']' && topChar != '[') ||
                		(ch == '}' && topChar != '{')) 
					{
               		 return false;
            		}
       		   }
           }

   		 // Check if the stack is empty at the end
   		 if(top==-1)
    		return true;
		}	
		
		char pop()
		{
			char c;
			if(top==-1)
				cout<<"\nStack is Empty....";
			else
			{
				c=stk[top];
				top--;
			}
			return c;
		}
			
		void push(char ch)
		{
			if(top==max-1)
				cout<<"\nStack Overflow....";
			else
			{
				top++;
				stk[top]=ch;
			}
		}

		
};

int main() 
{
    string exp;
    Expression e1;

    // Get input expression from the user
    cout << "Enter an expression: ";
    cin>>exp;

    // Check if the expression is well-parenthesized
    if (e1.Parenthesized(exp))
	 {
        cout << "The expression is well-parenthesized." <<endl;
    } 
	else 
	{
        cout << "The expression is not well-parenthesized." <<endl;
    }

    return 0;
}
